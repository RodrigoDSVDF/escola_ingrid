    <script>
        // --- CONSTANTES & CONFIGURAÇÕES INICIAIS ---
        let currentTab = 'multiplicacao';
        
        // --- LÓGICA DO TAB SWITCHER ---
        function switchTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));
            document.getElementById(`tab-${tabId}`).classList.remove('hidden');
            
            // Atualização dos estilos dos botões
            document.querySelectorAll('.tab-btn').forEach(btn => {
                btn.className = "tab-btn py-3 px-2 md:px-4 rounded-2xl font-bold text-xs md:text-base flex flex-col md:flex-row items-center justify-center gap-2 transition-all shadow-md bg-white text-slate-600 hover:bg-indigo-50";
            });
            
            const activeBtn = document.getElementById(`btn-tab-${tabId}`);
            if (tabId === 'multiplicacao') {
                activeBtn.className = "tab-btn py-3 px-2 md:px-4 rounded-2xl font-bold text-xs md:text-base flex flex-col md:flex-row items-center justify-center gap-2 transition-all shadow-md bg-indigo-600 text-white transform -translate-y-1";
            } else if (tabId === 'divisao') {
                activeBtn.className = "tab-btn py-3 px-2 md:px-4 rounded-2xl font-bold text-xs md:text-base flex flex-col md:flex-row items-center justify-center gap-2 transition-all shadow-md bg-teal-600 text-white transform -translate-y-1";
            } else if (tabId === 'quiz') {
                activeBtn.className = "tab-btn py-3 px-2 md:px-4 rounded-2xl font-bold text-xs md:text-base flex flex-col md:flex-row items-center justify-center gap-2 transition-all shadow-md bg-purple-600 text-white transform -translate-y-1";
            }
            
            currentTab = tabId;
        }

        // --- SISTEMA AUXILIAR DE CONTAR CASAS DECIMAIS ---
        function getDecimalPlaces(num) {
            const str = num.toString();
            if (str.indexOf('.') !== -1) {
                return str.split('.')[1].length;
            }
            if (str.indexOf(',') !== -1) {
                return str.split(',')[1].length;
            }
            return 0;
        }

        // --- LÓGICA DO LABORATÓRIO DE MULTIPLICAÇÃO ---
        function processarLaboratorioMult() {
            const input1 = document.getElementById('lab-m1').value.replace(',', '.');
            const input2 = document.getElementById('lab-m2').value.replace(',', '.');
            
            const f1 = parseFloat(input1);
            const f2 = parseFloat(input2);

            if (isNaN(f1) || isNaN(f2)) {
                return;
            }

            // Exibir o painel de passos
            const stepsBox = document.getElementById('mult-steps-box');
            stepsBox.classList.remove('hidden');

            const dp1 = getDecimalPlaces(f1);
            const dp2 = getDecimalPlaces(f2);
            const totalDp = dp1 + dp2;

            // Converter para inteiros
            const int1 = Math.round(f1 * Math.pow(10, dp1));
            const int2 = Math.round(f2 * Math.pow(10, dp2));
            const intResult = int1 * int2;
            const finalResult = (f1 * f2).toFixed(totalDp);

            // Passo 1 texto
            document.getElementById('step1-calc').innerHTML = `${int1} × ${int2} = <span class="text-indigo-600 font-extrabold">${intResult}</span>`;

            // Passo 2 lista
            document.getElementById('step2-list').innerHTML = `
                <li>O fator <strong>${f1.toString().replace('.', ',')}</strong> tem <span class="bg-blue-100 text-blue-800 px-2 py-0.5 rounded-full font-bold">${dp1} casa(s) decimais</span> à direita da vírgula.</li>
                <li>O fator <strong>${f2.toString().replace('.', ',')}</strong> tem <span class="bg-blue-100 text-blue-800 px-2 py-0.5 rounded-full font-bold">${dp2} casa(s) decimais</span> à direita da vírgula.</li>
                <li class="mt-2 font-bold text-slate-800 text-base">Soma das casas: ${dp1} + ${dp2} = <span class="bg-emerald-100 text-emerald-800 px-2 py-0.5 rounded-full font-extrabold">${totalDp} casa(s) no total!</span></li>
            `;

            // Passo 3 visualização interativa do movimento da vírgula
            const visualPath = document.getElementById('step3-visual-path');
            visualPath.innerHTML = '';
            
            // Construir o número sem vírgula como string
            const resultStr = intResult.toString();
            // Preencher com zeros se o número for menor do que o total de casas decimais necessárias
            let paddedStr = resultStr;
            while (paddedStr.length <= totalDp) {
                paddedStr = '0' + paddedStr;
            }

            const commaIndex = paddedStr.length - totalDp;
            
            // Renderização das letras/algarismos
            for (let i = 0; i < paddedStr.length; i++) {
                if (i === commaIndex) {
                    const commaSpan = document.createElement('span');
                    commaSpan.innerText = ',';
                    commaSpan.className = 'text-red-500 font-black text-4xl animate-bounce mx-0.5';
                    visualPath.appendChild(commaSpan);
                }
                const digitSpan = document.createElement('span');
                digitSpan.innerText = paddedStr[i];
                digitSpan.className = 'bg-slate-100 border-2 border-slate-200 rounded-lg px-2 py-1 text-slate-800 shadow-sm font-bold';
                visualPath.appendChild(digitSpan);
            }

            document.getElementById('step3-total-desc').innerText = `O resultado final é ${parseFloat(finalResult).toString().replace('.', ',')}!`;
            
            // Auto scroll suave para mostrar os passos
            stepsBox.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
        }


        // --- LÓGICA DO SIMULADOR DE DIVISÃO (CHOCOLATES) ---
        let simDividendo = 10;
        let simDivisor = 8;
        let simStep = 0; // 0: Start, 1: Distribute wholes, 2: Slice remainder (tenths), 3: Distribute tenths, 4: Slice next remainder (hundredths), 5: Distribute hundredths, 6: Finished
        let kidsShares = [];
        let curRemainder = 0;
        let visualChocolatesCount = 0;
        let decimalQuotientStr = "0";

        function presetDivisao(divd, divs) {
            document.getElementById('sim-dividendo').value = divd;
            document.getElementById('sim-divisor').value = divs;
            iniciarSimulacaoDivisao();
        }

        function iniciarSimulacaoDivisao() {
            simDividendo = parseInt(document.getElementById('sim-dividendo').value);
            simDivisor = parseInt(document.getElementById('sim-divisor').value);

            if (isNaN(simDividendo) || isNaN(simDivisor) || simDividendo < 1 || simDivisor < 1) {
                return;
            }

            document.getElementById('sim-active-board').classList.remove('hidden');
            
            simStep = 0;
            curRemainder = simDividendo;
            kidsShares = Array(simDivisor).fill(0);
            visualChocolatesCount = simDividendo;
            decimalQuotientStr = "";

            // Atualização inicial do algoritmo na caixa escura
            document.getElementById('math-dividendo').innerText = simDividendo;
            document.getElementById('math-divisor').innerText = simDivisor;
            document.getElementById('math-quociente').innerText = "0";
            document.getElementById('math-subtracao').innerText = "";

            renderChocolates(curRemainder, false);
            renderKids();

            definirPasso(1);
        }

        function renderChocolates(amount, isSlices, activeSlicesCount = 0) {
            const container = document.getElementById('sim-chocolate-display');
            container.innerHTML = '';
            
            if (amount === 0 && activeSlicesCount === 0) {
                container.innerHTML = '<span class="text-slate-400 font-bold text-sm">Sem sobras! Divisão concluída. 🎉</span>';
                return;
            }

            // Se for fatias, mostramos pedaços mais pequenos de chocolate
            if (isSlices) {
                for (let i = 0; i < activeSlicesCount; i++) {
                    const slice = document.createElement('div');
                    slice.className = 'w-6 h-10 bg-amber-700 rounded border-2 border-amber-900 shadow-sm flex items-center justify-center text-[10px] text-amber-200 font-bold';
                    slice.innerText = '•';
                    container.appendChild(slice);
                }
            } else {
                for (let i = 0; i < amount; i++) {
                    const choco = document.createElement('div');
                    choco.className = 'w-12 h-16 bg-amber-800 rounded-lg border-4 border-amber-950 shadow-md relative flex flex-col justify-between p-1 shrink-0';
                    choco.innerHTML = `
                        <div class="grid grid-cols-2 gap-0.5 h-full">
                            <div class="bg-amber-700 rounded-sm"></div>
                            <div class="bg-amber-700 rounded-sm"></div>
                            <div class="bg-amber-700 rounded-sm"></div>
                            <div class="bg-amber-700 rounded-sm"></div>
                        </div>
                    `;
                    container.appendChild(choco);
                }
            }
            
            // Etiqueta de quantidade restante
            document.getElementById('sim-resto-label').innerText = isSlices ? `${activeSlicesCount} fatia(s)` : `${amount} chocolate(s)`;
        }

        function renderKids() {
            const container = document.getElementById('sim-kids-display');
            container.innerHTML = '';
            
            const avatarList = ["👦", "👧", "👨", "👩", "👶", "👱", "👵", "👴"];

            for (let i = 0; i < simDivisor; i++) {
                const kid = document.createElement('div');
                kid.className = 'bg-white p-3 rounded-2xl border-2 border-slate-100 shadow-sm text-center flex flex-col items-center';
                
                const avatar = avatarList[i % avatarList.length];
                
                // Formatação do valor individual
                let valueText = kidsShares[i].toFixed(3).replace(/\.?0+$/, ''); // limpa zeros desnecessários
                if (valueText === "0") valueText = "Nada";
                else valueText = valueText.replace('.', ',') + " un";

                kid.innerHTML = `
                    <span class="text-3xl">${avatar}</span>
                    <span class="text-xs font-bold text-slate-500 mt-1">Neto ${i+1}</span>
                    <span class="text-xs bg-teal-100 text-teal-800 px-2 py-0.5 rounded-full font-bold mt-2">${valueText}</span>
                `;
                container.appendChild(kid);
            }
        }

        function definirPasso(stepNum) {
            simStep = stepNum;
            const btn = document.getElementById('btn-sim-action');
            const stepText = document.getElementById('sim-step-text');

            if (stepNum === 1) {
                stepText.innerHTML = `👋 <strong>Fase 1:</strong> Vamos dividir os chocolates inteiros por cada neto!`;

                btn.className = "bg-teal-600 hover:bg-teal-700 text-white font-bold px-5 py-2 rounded-xl text-sm transition-all shadow";

                btn.innerHTML = `<span>Fatiar Sobras (Pôr Vírgula)</span> <i class="fa-solid fa-scissors"></i>`;
                btn.className = "bg-amber-500 hover:bg-amber-600 text-white font-bold px-5 py-2 rounded-xl text-sm transition-all shadow animate-bounce";
            } else if (stepNum === 3) {
                stepText.innerHTML = `Distribui os décimos (fatias) pelos netos!`;
                btn.innerHTML = `<span>Distribuir Décimos</span> <i class="fa-solid fa-share"></i>`;
                btn.className = "bg-teal-600 hover:bg-teal-700 text-white font-bold px-5 py-2 rounded-xl text-sm transition-all shadow";

                btn.className = "bg-amber-500 hover:bg-amber-600 text-white font-bold px-5 py-2 rounded-xl text-sm transition-all shadow";

                stepText.innerHTML = `Distribui os centésimos pelos netos!`;
                btn.innerHTML = `<span>Distribuir Centésimos</span> <i class="fa-solid fa-share"></i>`;

                btn.className = "bg-slate-400 text-white font-bold px-5 py-2 rounded-xl text-sm cursor-not-allowed";
            }
        }

        function avancarPassoSimulacao() {
            if (simStep === 1) {
                // Divisão inteira
                const shareInt = Math.floor(curRemainder / simDivisor);
                if (shareInt > 0) {
                    for (let i = 0; i < simDivisor; i++) {
                        kidsShares[i] += shareInt;
                    }
                    const subtraido = shareInt * simDivisor;
                    curRemainder -= subtraido;

                    document.getElementById('math-subtracao').innerText = `- ${subtraido}`;
                    document.getElementById('math-quociente').innerText = shareInt;
                } else {
                    document.getElementById('math-quociente').innerText = "0";
                }

                renderChocolates(curRemainder, false);
                renderKids();

                if (curRemainder > 0) {
                    definirPasso(2);
                } else {
                    definirPasso(6);
                }
            } 
            else if (simStep === 2) {
                // Fatiar sobras: transforma unidades em décimos (unidade * 10)
                const fatias = curRemainder * 10;
                
                // Mostrar a vírgula no quociente matemático
                let currentQuociente = document.getElementById('math-quociente').innerText;
                decimalQuotientStr = currentQuociente + ",";
                document.getElementById('math-quociente').innerText = decimalQuotientStr;
                
                // No algoritmo escrito, pomos o 0 à direita do dividendo/resto
                document.getElementById('math-dividendo').innerText = fatias;
                document.getElementById('math-subtracao').innerText = "";

                renderChocolates(0, true, fatias);
                definirPasso(3);
            } 
            else if (simStep === 3) {
                // Distribuir fatias (décimos)
                const fatiasDisponiveis = curRemainder * 10;
                const shareDec = Math.floor(fatiasDisponiveis / simDivisor);
                
                for (let i = 0; i < simDivisor; i++) {
                    kidsShares[i] += (shareDec / 10);
                }

                const subtraido = shareDec * simDivisor;
                const restoFatias = fatiasDisponiveis - subtraido;
                curRemainder = restoFatias / 10;

                // Atualizar Quociente
                decimalQuotientStr += shareDec;
                document.getElementById('math-quociente').innerText = decimalQuotientStr;
                document.getElementById('math-dividendo').innerText = fatiasDisponiveis;
                document.getElementById('math-subtracao').innerText = `- ${subtraido}`;

                renderChocolates(0, true, restoFatias);
                renderKids();

                if (restoFatias > 0) {
                    definirPasso(4);
                } else {
                    definirPasso(6);
                }
            } 
            else if (simStep === 4) {
                // Centésimos: multiplicar o resto das fatias por 10 novamente
                const fatiasDisponiveis = curRemainder * 100; // centésimos
                
                document.getElementById('math-dividendo').innerText = fatiasDisponiveis;
                document.getElementById('math-subtracao').innerText = "";

                renderChocolates(0, true, fatiasDisponiveis);
                definirPasso(5);
            } 
            else if (simStep === 5) {
                // Distribuir centésimos
                const fatiasDisponiveis = curRemainder * 100;
                const shareCent = Math.floor(fatiasDisponiveis / simDivisor);

                for (let i = 0; i < simDivisor; i++) {
                    kidsShares[i] += (shareCent / 100);
                }

                const subtraido = shareCent * simDivisor;
                const restoCent = fatiasDisponiveis - subtraido;
                curRemainder = restoCent / 100;

                // Atualizar Quociente
                decimalQuotientStr += shareCent;
                document.getElementById('math-quociente').innerText = decimalQuotientStr;
                document.getElementById('math-dividendo').innerText = fatiasDisponiveis;
                document.getElementById('math-subtracao').innerText = `- ${subtraido}`;

                renderChocolates(0, true, restoCent);
                renderKids();

                definirPasso(6);
            }
        }

        function carregarNovaPergunta() {
            // Limpa o feedback anterior
            const feed = document.getElementById('quiz-feedback-box');
            feed.className = "hidden";
            document.getElementById('quiz-answer').value = "";

            const randomIndex = Math.floor(Math.random() * poolDeExercicios.length);
            activeQuestionObj = poolDeExercicios[randomIndex];

            document.getElementById('quiz-question').innerText = activeQuestionObj.q;
            document.getElementById('quiz-tip').innerText = `Dica: ${activeQuestionObj.tip}`;
        }

        function verificarQuiz() {
            const rawAns = document.getElementById('quiz-answer').value.trim().replace(',', '.');
            const targetAns = activeQuestionObj.a.replace(',', '.');
            
            const feed = document.getElementById('quiz-feedback-box');

            if (rawAns === "") {
                feed.className = "mt-6 p-4 rounded-2xl text-center font-bold text-sm md:text-base border-2 bg-amber-100 text-amber-800 border-amber-300 block";
                feed.innerHTML = "⚠️ Escreve uma resposta antes de tentares!";
                return;
            }

            if (parseFloat(rawAns) === parseFloat(targetAns)) {
                // Resposta Correta
                feed.className = "mt-6 p-4 rounded-2xl text-center font-bold text-sm md:text-base border-2 bg-emerald-100 text-emerald-800 border-emerald-300 block";
                feed.innerHTML = "🎉 Absolutamente correto! És um prodígio dos decimais! +10 Pontos";
                
                score += 10;
                streak++;
                
                triggerBadgeCheck();
            } else {
                // Resposta Errada
                feed.className = "mt-6 p-4 rounded-2xl text-center font-bold text-sm md:text-base border-2 bg-red-100 text-red-800 border-red-300 block";
                feed.innerHTML = `❌ Oh, quase lá! A resposta correta é ${activeQuestionObj.a.replace('.', ',')}. Olha a dica de rascunho de cima!`;
                
                streak = 0;
            }

            document.getElementById('score-counter').innerText = score;
            document.getElementById('streak-counter').innerText = streak;
            atualizarRank();
        }
